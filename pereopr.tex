\newcommand{\imgh}[3]
{
	\begin{figure}[hpt!]
		\center{\includegraphics[width=#1]{#2}}
		\caption{\small {#3}}
		\label{ris:#2}
	\end{figure}
}

\newcommand{\imgroot}[4]
{
	\begin{figure}[hpt!]
		\center{\includegraphics[angle=#4,width=#1]{#2}}
		\caption{\small {#3}}
		\label{ris:#2}
	\end{figure}
}
